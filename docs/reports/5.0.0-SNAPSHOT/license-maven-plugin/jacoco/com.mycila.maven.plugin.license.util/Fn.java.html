package com.mycila.maven.plugin.license.util;

import java.util.function.Supplier;

public class Fn {

  public static <T> Supplier<T> memoize(Supplier<T> supplier) {
    return new Supplier<T>() {
      private T value;
      private boolean resolved;

      @Override
      public T get() {
        if (!resolved) {
          value = supplier.get();
          resolved = true;
        }
        return value;
      }
    };
  }
}
